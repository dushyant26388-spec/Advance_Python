# 3. Create Fare_Per_Person
df["fare_per_person"]=df["Ticket_Price"] / df["Total_members"]

# 4. Create a column named Passenger_Type .
# Child (<18)
# Adult (18–59)
# Senior (60+)

def check_age(age):
  if age > 60:
    return "senior"
  elif age > 18:
    return "Adult"
  else:
    return "child"

df["Passenger_Type"] = df["Age"].apply(check_age)
df

# 5. Increase all ticket prices by 10%.
df["Ticket_Price"] = df["Ticket_Price"] *1.10
df

# 6. Give an additional 5% increase only to First Class passengers.
df.loc[df["Passenger_Class"]==1,"Ticket_Price"] *= 1.05
df

df=df.drop("Ticket",axis=1)  # this program is run in first but curerently not execute because Ticket column is deleted.

df.sort_values(by="Ticket_Price",ascending = False)

df.sort_values(by="Age",ascending=True)

# 10. Sort by Passenger Class and Ticket Price.
df.sort_values(by=["Passenger_Class","Ticket_Price"],ascending=[True,False])
