# Question 5 – Violin Plot (5 Marks)
# Create a violin plot showing the distribution of Lead Time for different Customer Types.

# Requirements

# Differentiate bookings using is_canceled .
# Add title and legend.

plt.figure(figsize=(10,6))
sns.violinplot(
    data=df,
    x="customer_type",
    y="lead_time",
    hue="is_canceled",
    palette="Set2",
    split=True
)

plt.title("Lead Time Distribution by Customer Type")
plt.xlabel("Customer Type")
plt.ylabel("Lead Time (Days)")
plt.legend(title="Is Canceled")
plt.show()
